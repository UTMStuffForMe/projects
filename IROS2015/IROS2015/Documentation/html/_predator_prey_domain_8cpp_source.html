395  for (int j=0; j<prey.size(); j++){

397  if (!prey[j]->is_caught){
398 // preyActions[j][0] = double(rand())/double(RAND_MAX)*2.0-1.0;
399 // preyActions[j][1] = double(rand())/double(RAND_MAX);
400  // escaping prey
401 
402  double fx = 0.0;
403  double fy = 0.0;
404  for (int i=0; i<predators.size(); i++){
405  double dx = predators[i]->x - prey[j]->x;
406  double dy = predators[i]->y - prey[j]->y;
407  double dxy = sqrt(dx*dx + dy*dy);
408  double Fpred=0;
409  if (dxy>0){
410  Fpred = 1/(dxy*dxy); // note: could multiply by charge force; omitting because assuming constant
411  fx += Fpred*(-dx/dxy); // F*xhat, also changes direction (opposite of predators)
412  fy += Fpred*(-dy/dxy); // F*yhat
413  }
414  }
415  double fDir = (atan2(fy,fx)/3.1416); //[-1,1]
416  fDir *= 0.5; // rescale to bounds [0,1]
417  if (fDir<0) fDir = 1.0-fDir; // rescale to bounds [0,1]
418 
419  // ABSOLUTELY SET ORIENTATION
