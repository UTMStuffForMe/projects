405  double max_3sig = ANodes[0]->GetMeanCTG() + 3*ANodes[0]->GetVarCTG() ;
406  double min_3sig = ANodes[0]->GetMeanCTG() - 3*ANodes[0]->GetVarCTG() ;
407  for (unsigned i = 0; i < ANodes.size(); i++)
408  {
409  if (max_3sig < ANodes[i]->GetMeanCTG()+3*ANodes[i]->GetVarCTG())

411  if (min_3sig > ANodes[i]->GetMeanCTG()-3*ANodes[i]->GetVarCTG())
412  min_3sig = ANodes[i]->GetMeanCTG()-3*ANodes[i]->GetVarCTG() ;
413  }

415  {
416  if (max_3sig < BNodes[i]->GetMeanCTG()+3*BNodes[i]->GetVarCTG())
417  max_3sig = BNodes[i]->GetMeanCTG()+3*BNodes[i]->GetVarCTG() ;
418  if (min_3sig > BNodes[i]->GetMeanCTG()-3*BNodes[i]->GetVarCTG())
419  min_3sig = BNodes[i]->GetMeanCTG()-3*BNodes[i]->GetVarCTG() ;
420  }
421 
422  // If code is taking too long, change n to a smaller value
423  // Note that since this is the numerical integration discretisation
424  // smaller n will provide coarser approximation to solution
425  int n = 100 ;
426  vector<double> x = linspace(min_3sig,max_3sig,n) ;
427  double dx = x[1]-x[0] ;
428  double pImprove = 0.0 ;
429  for (unsigned k = 0; k < x.size(); k++)
430  {
431  double p_cAi = 0.0 ;
432  for (unsigned i = 0; i < ANodes.size(); i++)
433  {
434  double mu_Ai = ANodes[i]->GetMeanCTG() ;
435  double sig_Ai = ANodes[i]->GetVarCTG() ;

437  double p_cA2 = 1.0 ;
438  for (unsigned j = 0; j < ANodes.size(); j++)

441  double sig_Aj = ANodes[j]->GetVarCTG() ;
442  if (j != i)
443  p_cA2 *= 0.5*easymath::erfc((x[k]-mu_Aj)/(sig_Aj*sqrt(2))) ;
444  }
445  p_cAi += p_cA1*p_cA2 ;
446  }
