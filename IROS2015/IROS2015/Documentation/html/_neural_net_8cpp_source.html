302 double NeuralNet::backProp(matrix1d &observations, matrix1d &t){
303  // 'observations' is the input vector, 't' is the 'target vector'
304  // returns the SSE for the output vector
305 
306  matrix2d Ohat; // outputs with bias
307  matrix3d D; // stored derivatives
308  // Go through network "feed forward" computation
309 
310  feedForward(observations,Ohat,D);
311 
312  matrix1d e(Ohat.back().size()-1,0.0); // "stored derivatives of the quadratic deviations"
313  for (uint i=0; i<Ohat.back().size()-1; i++){
314  e[i] = (Ohat.back()[i]-t[i]);
315  }
316 
317  // Hidden/output layer delta calcs
318  matrix2d delta;
319  for (int i=0; i<connections(); i++){
320  delta.push_back(matrix1d());
321  }
322  delta.back() = matrixMultiply(D.back(),e); // output layer delta
323 
324  for (int connection=connections()-2; connection>=0; connection--){ // back propagation
325  matrix2d mult = matrixMultiply(D[connection],W[connection+1]);
326  delta[connection]=matrixMultiply(mult,delta[connection+1]);
327  }
328 
329  // Corrections to weights
330  for (int connection=0; connection<connections(); connection++){
331  matrix2d DeltaWbarT = matrixMultiply(delta[connection],Ohat[connection]);
332  for (uint i=0; i<Wbar[connection].size(); i++){
333  for (uint j=0; j<Wbar[connection][i].size(); j++){
334  Wbar[connection][i][j] -= gamma_*DeltaWbarT[j][i]; // ji because it's transpose :)
335  }
336  }
337 
338  W[connection] = Wbar[connection];
339  W[connection].pop_back();
340  }
341 
342  // Calculate SS
343  return SSE(e);
344 }

347  Ohat.push_back(observations);
348  Ohat.back().push_back(1.0); // add 1 for bias
349

351  Ohat.push_back(matrixMultiply(Ohat[connection],Wbar[connection])); // Compute outputs
352  sigmoid(Ohat.back());
